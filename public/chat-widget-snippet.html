DOST R02 Chat Widget Snippet

Paste the code below before </body> on your Cagayan landing page. It adds a floating chat button that opens the chatbot in a popup.

If your chatbot is hosted at a different URL, replace https://dostr02chatbot.vercel.app in the iframe src with your URL.

<!-- DOST R02 Chatbot Widget -->
<div id="dost-chat-widget">
  <button type="button" id="dost-chat-fab" aria-label="Open chat">
    <svg xmlns="http://www.w3.org/2000/svg" width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
      <path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"></path>
    </svg>
  </button>
  <div id="dost-chat-popup">
    <div class="dost-chat-popup-header">
      <span>DOST R02 & Cagayan Assistant</span>
      <button type="button" id="dost-chat-close" aria-label="Close chat">&times;</button>
    </div>
    <iframe id="dost-chat-iframe" src="https://dostr02chatbot.vercel.app/embed" title="Chat"></iframe>
  </div>
</div>
<style>
  #dost-chat-widget { position: fixed; bottom: 24px; right: 24px; z-index: 9999; font-family: Inter, sans-serif; }
  #dost-chat-fab {
    width: 56px; height: 56px; border-radius: 50%; border: none; cursor: pointer;
    background: #e85c0d; color: #fff; box-shadow: 0 4px 20px rgba(232, 92, 13, 0.45);
    display: flex; align-items: center; justify-content: center; transition: transform 0.2s, box-shadow 0.2s;
  }
  #dost-chat-fab:hover { transform: scale(1.05); box-shadow: 0 6px 24px rgba(232, 92, 13, 0.5); }
  #dost-chat-popup {
    display: none; position: absolute; bottom: 72px; right: 0;
    width: 400px; height: 580px; max-width: calc(100vw - 48px); max-height: calc(100vh - 120px);
    background: #fff; border-radius: 16px; box-shadow: 0 10px 40px rgba(0,0,0,0.15);
    overflow: hidden; flex-direction: column;
  }
  #dost-chat-popup.is-open { display: flex; }
  .dost-chat-popup-header {
    padding: 12px 16px; background: #a82400; color: #fff; font-weight: 700; font-size: 14px;
    display: flex; align-items: center; justify-content: space-between; flex-shrink: 0;
  }
  #dost-chat-close { background: none; border: none; color: #fff; font-size: 24px; cursor: pointer; line-height: 1; padding: 0 4px; }
  #dost-chat-iframe { flex: 1; width: 100%; border: none; min-height: 480px; }
</style>
<script>
  (function() {
    var fab = document.getElementById('dost-chat-fab');
    var popup = document.getElementById('dost-chat-popup');
    var close = document.getElementById('dost-chat-close');
    if (fab && popup) {
      fab.addEventListener('click', function() { popup.classList.toggle('is-open'); });
      if (close) close.addEventListener('click', function() { popup.classList.remove('is-open'); });
    }
  })();
</script>
